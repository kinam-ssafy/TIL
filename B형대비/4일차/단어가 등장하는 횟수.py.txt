EXPONENT1 = 32
EXPONENT2 = 37
EXPONENT3 = 41

def getCount(s, pattern):
    count = 0

    stringLength = len(s)
    patternLength = len(pattern)

    stringHash1 = 0
    patternHash1 = 0

    stringHash2 = 0
    patternHash2 = 0

    stringHash3 = 0
    patternHash3 = 0

    power1 = 1
    power2 = 1
    power3 = 1

    for i in range(stringLength - patternLength + 1):   # Window's starting position
        # Check if string[i ... i + patternLength - 1] is equal to pattern
        if i == 0:
            for j in range(patternLength):
                stringHash1 += ord(s[patternLength - 1 - j]) * power1
                patternHash1 += ord(pattern[patternLength - 1 - j]) * power1

                stringHash2 += ord(s[patternLength - 1 - j]) * power2
                patternHash2 += ord(pattern[patternLength - 1 - j]) * power2

                stringHash3 += ord(s[patternLength - 1 - j]) * power3
                patternHash3 += ord(pattern[patternLength - 1 - j]) * power3

                if j < patternLength - 1:
                    power1 *= EXPONENT1;
                    power2 *= EXPONENT2;
                    power3 *= EXPONENT3;
        else:
            stringHash1 = EXPONENT1 * (stringHash1 - ord(s[i - 1]) * power1) + ord(s[patternLength - 1 + i])
            stringHash2 = EXPONENT2 * (stringHash2 - ord(s[i - 1]) * power2) + ord(s[patternLength - 1 + i])
            stringHash3 = EXPONENT3 * (stringHash3 - ord(s[i - 1]) * power3) + ord(s[patternLength - 1 + i])

        if stringHash1 == patternHash1 and stringHash2 == patternHash2 and stringHash3 == patternHash3:
            count += 1

    return count

T = int(input())

for testCase in range(1, T + 1):
    B = input()
    S = input()

    result = getCount(B, S)

    print(f'#{testCase} {result}')
