class Heap:
    def __init__(self):
        self.arr = [0 for i in range(100001)]
        self.cnt = 0

    def getParent(self, child):
        return child // 2

    def getLeft(self, parent):
        return parent * 2

    def getRight(self, parent):
        return parent * 2 + 1

    def heapify(self):
        now = 1
        while self.getRight(now) <= self.cnt:
            larger = now
            left = self.getLeft(now)
            right = self.getRight(now)
            if self.arr[left] > self.arr[larger]:
                larger = left
            if self.arr[right] > self.arr[larger]:
                larger = right

            if larger != now:
                self.arr[now], self.arr[larger] = self.arr[larger], self.arr[now]
                now = larger
            else:
                break

    def add(self, data):
        self.cnt += 1
        self.arr[self.cnt] = data
        now = self.cnt
        while now > 1:
            parent = self.getParent(now)
            if self.arr[now] > self.arr[parent]:
                self.arr[now], self.arr[parent] = self.arr[parent], self.arr[now]
                now = parent
            else:
                break

    def poll(self):
        max_val = self.arr[1]
        self.arr[1] = self.arr[self.cnt]
        self.arr[self.cnt] = 0
        self.cnt -= 1
        self.heapify()
        return max_val

    def isEmpty(self):
        return self.cnt == 0

TC = int(input())
for tc in range(1, TC + 1):
    N = int(input())

    heap = Heap()
    print(f'#{tc} ', end='')

    for i in range(N):
        input_line = input().split()

        if input_line[0] == '1':
            heap.add(int(input_line[1]))
        else:
            if heap.isEmpty():
                print(-1, end=' ')
            else:
                print(heap.poll(), end=' ')

    print()
