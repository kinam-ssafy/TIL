testcase = int(input())
total = (1 << 10) - 1

for i in range(1, testcase + 1):
    N = int(input())

    visited = 0
    count = 0
    while True:
        count += 1
        strNum = str(N * count)
        for c in strNum:
            num = int(c)
            visited |= (1 << num)

        if visited == total:
            break

    print(f'#{i} {N * count}')
